

# async，future
- 提供了一种比直接使用 std::thread 更高级、更方便的异步并发编程模型。
```cpp
#include <iostream>
#include <future>
using namespace std;
int func(){
    int i=0;
    while(i<1000){
        i++;
        cout<<i<<endl;
    }
    return i;
}

int main(){
    future<int> result = async(func); //无策略异步执行
    future<int> result2 = async(launch::async,func); //强制异步执行
    future<int> result3 = async(launch::deferred,func); //延迟执行，只有在调用get()或wait()时才会执行
    result.get();
    return 0;
}
```
# packaged_task
- 是一个模板类，它是对一个任务（函数、Lambda等）的包装，不会执行
```cpp
#include<iostream>
#include <future>
using namespace std;
int func(){
    int i=0;
    while(i<1000){
        i++;
        cout<<i<<endl;
    }
    return i;
}
int main(){
    //1. 包装任务：定义该任务返回值为int，参数列表为空
    packaged_task<int()> task(func); 
    //2. 获取future对象，get_future() 的核心用处：它允许我们通过调用 future 对象的 get() 方法获取任务的返回值。
    future<int> result = task.get_future(); 
    //3. 启动线程执行任务
    thread t(move(task)); 
    result.get(); //等待任务完成并获取结果
}
```
# promise
- promise 是一个模板类，用于在一个线程中设置一个值，然后在另一个线程中获取这个值。
- 它与 future和async配合使用，提供了一种在线程之间传递结果的机制。
```cpp
#include <iostream>
#include <future>
#include <thread>
using namespace std;

// 语法：传引用 std::promise<int>& p
void func(promise<int>& p) {
    int sum = 0;
    for(int i = 1; i <= 10; i++) sum += i;
    
    // 算完后，把结果塞进 promise 对象里
    p.set_value(sum); 
}

int main() {
    promise<int> prom; // 声明一个包裹 int 的 promise
    future<int> fut = prom.get_future(); 
    
    // 注意：用 std::ref 按引用包一下，再传给线程函数
    thread t(func, ref(prom)); 

    cout << "结果是: " << fut.get() << endl; 
    
    t.join();
    return 0;
}
