#
由于两个线程同时对a进行++操作，导致数据共享问题，最终输出的结果不确定。 
```cpp
#include <thread>
#include <iostream>
using namespace std;

int a = 0;
void test(){
    for(int i = 0; i < 1000000; i++){
        a++;
    }
    
}
int main() {

    thread t( test);
    thread t1( test);
    t1.join();
    t.join();
    cout << a << endl;
    return 0;
}
```
# 互斥锁
```cpp
#include <thread>
#include <iostream>
#include <mutex>// 引入互斥锁头文件
using namespace std;

int a = 0;
mutex m;// 定义互斥锁对象
void test(){
    for(int i = 0; i < 1000000; i++){
        m.lock(); // 加锁,在解锁前其他线程无法访问a
        a++;
        m.unlock(); // 解锁
    }
    
}
int main() {

    thread t( test);
    thread t1( test);
    t1.join();
    t.join();
    cout << a << endl;
    return 0;
}
